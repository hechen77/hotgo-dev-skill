## 7.9 API 层：utils/request.js + api/index.js

### 请求封装

`utils/request.js`：

```js
import { getToken, delToken, getUserInfoName } from './token'
import { baseUrl } from './config'

/**
 * 全局请求封装（默认导出函数）
 * @param {string} path   请求路径
 * @param {string} method 请求方法（GET/POST）
 * @param {object} data   请求体
 * @param {object} header 自定义请求头
 * @param {boolean} loading 是否显示加载中
 */
export default function request(path, method, data = {}, header = {}, loading = true) {
  const token = getToken()

  if (loading) {
    uni.showLoading({ title: '加载中', mask: true })
  }

  return new Promise((resolve, reject) => {
    header.Authorization = token

    uni.request({
      url: baseUrl() + path,
      method: method,
      data: data,
      timeout: 60000,
      header,
      success(response) {
        if (response.data.code === 61) {
          // Token 过期
          delToken()
          delToken(getUserInfoName())
          if (loading) {
            uni.showToast({ title: '验证过期，请重新登录', icon: 'error' })
          }
        } else if (response.data.code !== 0) {
          if (loading) {
            uni.showToast({ title: response.data.message, icon: 'error' })
          }
        }
        resolve(response.data)
      },
      fail(err) {
        uni.showToast({ title: '服务响应失败', icon: 'none' })
        reject(err)
      },
      complete() {
        if (loading) {
          uni.hideLoading()
        }
      }
    })
  })
}
```

**关键约定**：
- 使用**默认导出**（`export default function request`），不是具名导出
- 请求路径是相对路径，由 `baseUrl()` 拼接完整 URL
- Token 自动从 `getToken()` 获取并注入 Header
- `code === 61` 时自动清除 Token（登录过期）
- `code !== 0` 时弹 Toast 提示错误

## 7.10 Token 管理：utils/token.js

```js
const tokenName = 'access_token'
const userInfoName = 'app_user_info'

// 获取 Token
export function getToken(name) {
  return uni.getStorageSync(name || tokenName)
}

// 设置 Token
export function setToken(val, name) {
  uni.setStorageSync(name || tokenName, val)
}

// 删除 Token
export function delToken(name) {
  uni.removeStorageSync(name || tokenName)
}

// 获取 Token 存储键名
export function getTokenName() {
  return tokenName
}

// 获取用户信息存储键名
export function getUserInfoName() {
  return userInfoName
}

// 判断是否登录
export function isLogin() {
  return !!getToken()
}

// 退出登录
export function logout() {
  delToken()
  delToken(getUserInfoName())
  uni.reLaunch({ url: '/pages/login/login' })
}
```

**设计要点**：
- `token.js` **独立于** `config.js`，各司其职
- 使用 `uni.getStorageSync/setStorageSync` 同步读写
- `getToken(name)` 接受可选参数，不传则用默认键名
- `isLogin()` 只检查 Token 是否存在

## 7.11 多环境配置：utils/config.js

```js
const localServer = '192.168.1.81:8000'
const devServer = 'dev.example.com'
const prodServer = 'api.example.com'
const env = 'prod'           // ⚠️ 发布时切换：local | dev | prod
const wsSuffix = '/socket?authorization='

export function baseUrl() {
  if (env === 'dev')  return `https://${devServer}`
  if (env === 'prod') return `https://${prodServer}`
  return `http://${localServer}`
}

export function wsUrl() {
  const token = require('./token').getToken()
  if (env === 'dev')  return `wss://${devServer}${wsSuffix}${token}`
  if (env === 'prod') return `wss://${prodServer}${wsSuffix}${token}`
  return `ws://${localServer}${wsSuffix}${token}`
}

// WebSocket 心跳间隔（3分钟）
export const wsHeartBeatTime = 3 * 60 * 1000
```

**设计要点**：
- `env` 变量控制环境切换（发布时只改一行）
- `baseUrl()` 被 `utils/request.js` 调用拼装完整请求 URL
- `wsUrl()` 在 `utils/websocket.js` 连接时调用，Token 直接从 `token.js` 获取
- `wsHeartBeatTime` 作为常量导出

## 7.12 WebSocket：utils/websocket.js

```js
import { wsHeartBeatTime, wsUrl } from './config'
import { getToken } from './token'

let socketTask = null
let heartbeatInterval = null
let reconnectTimeout = null
let isOpenSocket = false

// 初始化 WebSocket
export function ws_init() {
  isOpenSocket = false
  heartbeatInterval = null
  reconnectTimeout = null

  try {
    return ws_connect()
  } catch (e) {
    isOpenSocket = false
    ws_reconnect()
  }
}

// 建立连接
export function ws_connect() {
  if (!getToken()) return  // 未登录不连接

  socketTask = uni.connectSocket({
    url: wsUrl(),
    success: () => console.log('正在建立 WebSocket 连接...')
  })

  socketTask.onOpen(() => {
    console.log('WebSocket 连接成功')
    clearTimeout(reconnectTimeout)
    clearInterval(heartbeatInterval)
    ws_startHeartbeat()
    isOpenSocket = true
  })

  socketTask.onClose(() => {
    isOpenSocket = false
    ws_reconnect()
  })

  socketTask.onMessage((res) => {
    // 消息交给 ws_handleMessage 处理
    ws_handleMessage(res.data)
  })
}

// 心跳
function ws_startHeartbeat() {
  heartbeatInterval = setInterval(() => {
    ws_send(JSON.stringify({ event: 'ping' }))
  }, wsHeartBeatTime)
}

// 发送消息
export function ws_send(data) {
  if (socketTask) {
    socketTask.send({ data, success: () => console.log('消息发送成功') })
  }
}

// 重连（3 秒后）
export function ws_reconnect() {
  if (socketTask) ws_close()
  clearInterval(heartbeatInterval)
  if (!isOpenSocket) {
    reconnectTimeout = setTimeout(() => ws_connect(), 3000)
  }
}

// 关闭连接
export function ws_close(manualClose = false) {
  if (manualClose) isOpenSocket = true  // 手动关闭不重连
  if (socketTask) socketTask.close()
}
```

**设计要点**：
- 使用 `uni.connectSocket`（Uni-app 原生 API）
- 通过 `isOpenSocket` 标志区分"手动关闭"和"被动断开"
- 手动关闭（如退出登录）不触发重连
- 被动断开 3 秒后自动重连
- Token 从 URL 参数传递（`?authorization=xxx`）

### 消息处理

消息处理单独提取为函数，通过 `switch(event)` 分发：

```js
export function ws_handleMessage(data) {
  const { event, data: payload } = JSON.parse(data)
  switch (event) {
    case 'kick':
      ws_close(true)
      logout()
      break
    case 'refresh':
      // 触发页面刷新
      break
    default:
      console.log('未处理的消息:', event)
  }
}
```

## 7.13 加密工具：utils/encrypt.js

```js
import CryptoJS from 'crypto-js'

const defaultKey = 'your_16_byte_key!'  // ⚠️ 16 字节密钥

export const aesEcb = {
  // AES-ECB 加密
  encrypt(word, keyStr = defaultKey) {
    const key = CryptoJS.enc.Utf8.parse(keyStr)
    const src = CryptoJS.enc.Utf8.parse(word)
    const encrypted = CryptoJS.AES.encrypt(src, key, {
      mode: CryptoJS.mode.ECB,
      padding: CryptoJS.pad.Pkcs7
    })
    return encrypted.toString()
  },

  // AES-ECB 解密
  decrypt(word, keyStr = defaultKey) {
    const key = CryptoJS.enc.Utf8.parse(keyStr)
    const decrypt = CryptoJS.AES.decrypt(word, key, {
      mode: CryptoJS.mode.ECB,
      padding: CryptoJS.pad.Pkcs7
    })
    return CryptoJS.enc.Utf8.stringify(decrypt).toString()
  }
}
```

                     image="search" tip="暂无数据" />

      <view v-for="item in list" :key="item.id" class="card"
            @click="goDetail(item.id)">
        <text class="card-title">{{ item.title }}</text>
        <text class="card-desc">{{ item.description }}</text>
      </view>
    </view>

<script setup>
import { ref, onMounted } from 'vue'
import { getList } from '@/api/index'
import { onReachBottom, onPullDownRefresh } from '@dcloudio/uni-app'

const list = ref([])
const loading = ref(false)
const loadState = ref('finished')  // finished | loading | error
const page = ref(1)

async function loadData(isRefresh = false) {
  if (loading.value) return
  loading.value = true
  loadState.value = 'loading'

  if (isRefresh) {
    page.value = 1
    list.value = []
  }

  try {
    const res = await getList({ page: page.value, pageSize: 10 })
    list.value.push(...(res.list || []))
    loadState.value = res.page < res.pageCount ? 'finished' : 'finished'
  } catch (e) {
    loadState.value = 'error'
  } finally {
    loading.value = false
  }
}

onMounted(() => loadData(true))

onReachBottom(() => {
  page.value++
  loadData()
})

onPullDownRefresh(async () => {
  await loadData(true)
  uni.stopPullDownRefresh()
})

function goDetail(id) {
  uni.navigateTo({ url: `/pages/detail/detail?id=${id}` })
}
</script>

## 7.20 与 Web 端和移动端的核心差异

| 特性 | 微信小程序 | Web 管理后台 | Flutter 移动端 |
|------|----------|-------------|--------------|
| **框架** | Uni-app (Vue 3) | Vue 3 + Vite | Flutter + Dart |
| **UI 组件** | wot-design-uni | Naive UI | TDesign Flutter |
| **路由** | `pages.json` 配置 | Vue Router (hash) | GetX Router |
| **HTTP** | `uni.request` | `http.request` (Axios) | `HttpManager` (Dio) |
| **存储** | `uni.getStorageSync` | `localStorage` | `FlutterSecureStorage` |
| **WebSocket** | `uni.connectSocket` | `new WebSocket()` | `web_socket_channel` |
| **导航** | 页面栈 `navigateTo/redirectTo` | 侧边栏菜单 | BottomNavigationBar |
| **下拉刷新** | `onPullDownRefresh` | — | `RefreshIndicator` |
| **包大小** | 主包 ≤ 2MB | 无限制 | 无限制 |
| **后台运行** | 挂起 | 支持 | Foreground Service |
| **条件编译** | `#ifdef MP-WEIXIN` | 不需要 | `Platform.isAndroid` |
| **CSS 单位** | rpx（750 = 屏宽） | px / rem | 逻辑像素 |
